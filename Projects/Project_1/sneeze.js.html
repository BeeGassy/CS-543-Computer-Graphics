200  function PointDrawer() {
201  
202      gl.clear(gl.COLOR_BUFFER_BIT);
203  
204      //All for setting up the points
205      //
206      var vBuffer = gl.createBuffer();
207      gl.bindBuffer(gl.ARRAY_BUFFER, vBuffer);
208  
209      for (var i = 0; i < XandYArr.length; i++) {
210          var currentsegment = XandYArr[i];
211  
212          gl.bufferData(gl.ARRAY_BUFFER, flatten(currentsegment), gl.STATIC_DRAW);
213          var jvPosition = gl.getAttribLocation(program, "vPosition");
214          gl.vertexAttribPointer(jvPosition, 2, gl.FLOAT, false, 0, 0);
215  
216          gl.enableVertexAttribArray(jvPosition);
217  
218          setColor();
219          gl.drawArrays(gl.LINE_STRIP, 0, currentsegment.length);
220      }
221  
222  }

253  //setViewPort takes in the left, right, bottom and top of the extent parameters and uses those extent
254  //parameters to draw the width and the height which help create both the update height and width and the gl.viewport
255  function setViewPort(left, right, bottom, top) {
256  
257      var width;
258      width = right - left;
259  
260      var height;
261      height = top - bottom;
262  
263      var UpdatedHeight;
264      var UpdatedWidth;
265  
266      if (height < width) {
267          console.log("width was larger than the height");
268  
269          //when the limiting factor is the width
270          UpdatedHeight = ((height * canvas.width) / width);
271          gl.viewport(0, 0, canvas.width, UpdatedHeight);
272      } else if (height > width) {
273          console.log("height was larger than the width");
274  
275          //when the limiting factor is the height
276          UpdatedWidth = ((width * canvas.height) / height);
277          gl.viewport(0, 0, UpdatedWidth, canvas.height);
278      } else {
279          console.log("Height and width are the same size and do not need to be updated");
280          gl.viewport(0, 0, canvas.width, canvas.height);
281      }
282  }

326  function FileFormatter(FileContentsInput) {
327      console.log("entered FileFormatter");
328  
329      var fc;
330      fc = FileContentsInput;
331  
332      //splitting the contents of the file into different lines that can be iterated through
333      lines = fc.split('\n');
334  
335      // Removes empty lines
336      lines = lines.filter(FilterHelper);
337  
338      //extent dimensions that are originally set to default values unless otherwise changed
339      var leftd;
340      leftd = parseFloat(0);
341      ExtentArr[0] = leftd;
342  
343      var topd;
344      topd = parseFloat(480);
345      ExtentArr[1] = topd;
346  
347      var rightd;
348      rightd = parseFloat(640);
349      ExtentArr[2] = rightd;
350  
351      var bottomd;
352      bottomd = parseFloat(0);
353      ExtentArr[3] = bottomd;
354  
355      //return value of ortho which will be multiplied to vposition later on
356      var ProjMatrix;
357  
358      StringParser(fc);
359  
360      //lines needed to find the Length of First Line (LoFl)
361      var LoFL;
362      var brrr = lines[0].trim().split("  ").join(" ").split(" ");
363  
364  
365      LoFL = brrr.length;
366      if (LoFL == 1) {
367  
368          setViewPort(leftd, rightd, bottomd, topd);
369  
370          //left, right, bottom, top, near, far
371          ProjMatrix = ortho(leftd, rightd, bottomd, topd, -1.0, 1.0);
372      } else {
373  
374          setViewPort(ExtentArr[0], ExtentArr[2], ExtentArr[3], ExtentArr[1]);
375  
376          //left, right, bottom, top, near, far
377          ProjMatrix = ortho(ExtentArr[0], ExtentArr[2], ExtentArr[3], ExtentArr[1], -1.0, 1.0);
378      }
379  
380      //sending the ProjMatrix to the fragment shader
381      var projMatrixLoc = gl.getUniformLocation(program, "ProjMatrix");
382      gl.uniformMatrix4fv(projMatrixLoc, false, flatten(ProjMatrix));
383      PointDrawer();
384  }

393  function StringParser(FileContentsInput) {
394      console.log("entered StringParser");
395  
396      var words;
397      words = [];
398  
399      //extent dimensions that are originally set to default values unless otherwise changed
400      var leftd;
401      var topd;
402      var rightd;
403      var bottomd;
404  
405      //Number of points in the polyline
406      var NoPiP;
407  
408      var temparr;
409  
410      //iterates through lines
411      for (var i = 0; i < lines.length; i++) {
412          console.log("line: %d", i);
413  
414          //checks to see if "file" or any amount of "*" is in the line
415          if (!(lines[i].match(/file/i) || lines[i].match(/\*+/))) {
416  
417              //words is the array of words making up the line
418              words = lines[i].trim().split("  ").join(" ").split(" ");
419  
420              //checks to see the line is the "extent" parameters
421              if (words.length > 2) {
422                  console.log("iterating through extent parameters");
423  
424                  //applies the current extent parameter to the extent array
425                  ExtentArr[0] = parseFloat(words[0]); //left
426                  var left = ExtentArr[0];
427  
428                  ExtentArr[1] = parseFloat(words[1]); //top
429                  var top = ExtentArr[1];
430  
431                  ExtentArr[2] = parseFloat(words[2]); //right
432                  var right = ExtentArr[2];
433  
434                  ExtentArr[3] = parseFloat(words[3]); //bottom
435                  var bottom = ExtentArr[3];
436  
437              } //checks for either NoP or NoPiP
438              else if (words.length == 1) {
439                  console.log("checking NoP or the NoPiP");
440  
441                  if (NoP == undefined) {
442                      NoP = words[0];
443  
444  
445                  } else {
446                      NoPiP = words[0];
447  
448                      if (temparr != undefined) {
449                          XandYArr.push(temparr);
450                      }
451                      temparr = [];
452                  }
453  
454  
455              } //checks for x and y coordinates
456              else if (words.length == 2) {
457                  console.log("checking x and y coordinates");
458  
459                  //applies the x and y coordinates into the array
460                  temparr.push(vec2(parseFloat(words[0]), parseFloat(words[1])));
461  
462              } else {
463                  console.log("An Error Occurred within the word parser");
464                  break;
465              }
466  
467          } else {
468              console.log("currently going past the unneeded lines");
469          }
470      }
471      XandYArr.push(temparr);
472  }

475  function RenderDraw() {
476      gl.clear(gl.COLOR_BUFFER_BIT);
477  
478      //All for setting up the points
479      //
480      console.log(DrawnPoints);
481  
482      for (var i = 0; i < DrawnPoints.length; i++) {
483          //create and stores points to buffer
484          var vBuffer = gl.createBuffer();
485          gl.bindBuffer(gl.ARRAY_BUFFER, vBuffer);
486          gl.bufferData(gl.ARRAY_BUFFER, flatten(DrawnPoints[i]), gl.STATIC_DRAW);
487  
488          //applies points to vposition
489          var jvPosition = gl.getAttribLocation(program, "vPosition");
490          gl.vertexAttribPointer(jvPosition, 2, gl.FLOAT, false, 0, 0);
491          gl.enableVertexAttribArray(jvPosition);
492  
493          //depending on how many points there are there will be a line made or a point
494          if (DrawnPoints[i].length == 1) {
495              gl.drawArrays(gl.POINTS, 0, DrawnPoints[i].length);
496          } else {
497              gl.drawArrays(gl.LINE_STRIP, 0, DrawnPoints[i].length);
498          }
499
